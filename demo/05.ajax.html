$("#form0").on("submit",function(e){
    /*该方法执行需要f2e-server服务器支持*/
    $.ajax({
        url: "/upload?target=demo/data/upload.json&uploadUrl=demo/data/",
        type: "post",
        form: this,
        success: function( json ){
            console.log( json );
        },
        error: function(xhr){
            alert( 'wrong' )
        }
    });
    e.preventDefault();
});
